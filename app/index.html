void polygon(float x,float y,float r,int n){
  pushMatrix();
  translate(x, y);
  scale(1, -1);
  rotate(HALF_PI);
  beginShape();
  for (int i = 0; i < n; i++) {
   vertex(r*cos(TWO_PI * i/n),r*sin(TWO_PI * i/n));
  }
  endShape(CLOSE);
  popMatrix();
}
